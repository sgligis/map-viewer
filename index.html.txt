  <script>
    require(["esri/Map", "esri/views/MapView", "esri/layers/GeoJSONLayer"], 
    (Map, MapView, GeoJSONLayer) => {
      
      const map = new Map({ basemap: "streets" });
      
      const view = new MapView({
        container: "viewDiv",
        map: map,
        center: [75.7873, 26.9124], // Rajasthan
        zoom: 12
      });

      // Sample GeoJSON URL (replace with yours)
      const riicoLayer = new GeoJSONLayer({
        url: "https://raw.githubusercontent.com/giswqs/geojson-examples/master/india/rajasthan.geojson",
        popupTemplate: {
          title: "RIICO Plot",
          content: "This is a sample plot"
        }
      });

      map.add(riicoLayer);
    });
  </script>
